181 inline static bool DragFloatN_Colored(
182 const char* label,
183 float* v,
184 int components,
185 float v_speed,
186 float v_min,
187 float v_max,
188 const char* display_format,
189 float power,
190 ImVec4 x_axis,
191 ImVec4 y_axis,
192 ImVec4 z_axis
193 ) {
194 ImGuiWindow* window = ImGui::GetCurrentWindow();
195 if (window->SkipItems)
196 return false;
197
198 ImGuiContext& g = *GImGui;
199 bool value_changed = false;
200 ImGui::BeginGroup();
201 ImGui::PushID(label);
202 ImGui::PushMultiItemsWidths(components, ImGui::CalcItemWidth());
203
204 unsigned int R = SolidUtils::ColorVecToIMU32(x_axis);
205 unsigned int G = SolidUtils::ColorVecToIMU32(y_axis);
206 unsigned int B = SolidUtils::ColorVecToIMU32(z_axis);
207
208 for (int i = 0; i < components; i++) {
209 static const ImU32 colors[] = {
210 R, // these are read backwards but the bits are still read left to right
211 G, // 00ff6a
212 B, // 0088ff
213 0xBBFFFFFF, // white
214 };
215
216 ImGui::PushID(i);
217 value_changed |= ImGui::DragFloat("##v", &v[i], v_speed, v_min, v_max, display_format, power);
218
219 const ImVec2 min = ImGui::GetItemRectMin();
220 const ImVec2 max = ImGui::GetItemRectMax();
221 const float spacing = g.Style.FrameRounding;
222 const float halfSpacing = spacing / 2;
223
224 window->DrawList->AddLine({ min.x + spacing, max.y - halfSpacing }, { max.x - spacing, max.y - halfSpacing }, colors[i], 1);
225
226 ImGui::SameLine(0, g.Style.ItemInnerSpacing.x);
227 ImGui::PopID();
228 ImGui::PopItemWidth();
229 }
230
231 ImGui::PopID();
232 ImGui::TextUnformatted(label, ImGui::FindRenderedTextEnd(label));
233 ImGui::EndGroup();
234
235 return value_changed;

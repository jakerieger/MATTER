140 inline static unsigned int LoadTexture(const char* path) {
141 unsigned int textureID;
142 glGenTextures(1, &textureID);
143
144 int width, height, nrComponents;
145 unsigned char* data = stbi_load(path, &width, &height, &nrComponents, 0);
146 if (data) {
147 GLenum format;
148 if (nrComponents == 1)
149 format = GL_RED;
150 else if (nrComponents == 3)
151 format = GL_RGB;
152 else if (nrComponents == 4)
153 format = GL_RGBA;
154
155 glBindTexture(GL_TEXTURE_2D, textureID);
156 glTexImage2D(GL_TEXTURE_2D, 0, format, width, height, 0, format, GL_UNSIGNED_BYTE, data);
157 glGenerateMipmap(GL_TEXTURE_2D);
158
159 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
160 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
161 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
162 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
163
164 stbi_image_free(data);
165 } else {
166 printf("Failed to load texture");
167 stbi_image_free(data);
168 }
169
170 return textureID;
171 }

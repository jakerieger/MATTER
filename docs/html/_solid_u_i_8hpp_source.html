139 inline static void LoadColors(std::string themeName) {
140 std::string defaultTheme = SolidUtils::GetEditorThemesPath() + "\\" + themeName + ".xml";
141 pugi::xml_document themeDoc;
142 pugi::xml_parse_result result = themeDoc.load_file(defaultTheme.c_str());
143
144 if (!result)
145 SolidLogger::GetInstance()->Log(("Failed to load theme: " + themeName).c_str(), defaultTheme.c_str(), __FILE__, (int*)__LINE__, LogLevel::LogLevel_ERROR);
146
147 for (pugi::xml_node color: themeDoc.child("theme").child("colors").children("color")) {
148 std::string name = color.attribute("name").value();
149 std::string hex = color.attribute("value").value();
150 float alpha = color.attribute("opacity").as_float();
151
152 ImVec4 colorVec = SolidUtils::ColorVecFromHex(hex.c_str(), alpha);
153 colors[name] = colorVec;
154 }
155
156 pugi::xml_node window = themeDoc.child("theme").child("window");
157 window_rounding = window.attribute("window_rounding").as_float();
158 frame_rounding = window.attribute("frame_rounding").as_float();
159 border_size = window.attribute("border_width").as_float();
160 }

252 inline static void ParseProjectAssets(SolidProject &project) {
253 project_assets.textures.clear();
254 project_assets.meshes.clear();
255 project_assets.materials.clear();
256 project_assets.fonts.clear();
257 project_assets.scripts.clear();
258 project_assets.scenes.clear();
259 project_assets.game_objects.clear();
260 project_assets.audio.clear();
261 project_assets.fonts.clear();
262
263 for (const auto &entry : std::filesystem::directory_iterator(project.GetProjectPath())) {
264 SolidUtils::FileType type = SolidUtils::GetFileType(entry.path().extension().string());
265
266 switch (type) {
267 case SolidUtils::FileType::FileType_Font:
268 project_assets.fonts.push_back(entry.path());
269 break;
270 case SolidUtils::FileType::FileType_Script:
271 project_assets.scripts.push_back(entry.path());
272 break;
273 case SolidUtils::FileType::FileType_Scene:
274 project_assets.scenes.push_back(entry.path());
275 break;
276 case SolidUtils::FileType::FileType_Unknown:
277 project_assets.game_objects.push_back(entry.path());
278 break;
279 case SolidUtils::FileType::FileType_Texture:
280 project_assets.textures.push_back(entry.path());
281 break;
282 case SolidUtils::FileType::FileType_Model:
283 project_assets.meshes.push_back(entry.path());
284 break;
285 case SolidUtils::FileType::FileType_Material:
286 project_assets.materials.push_back(entry.path());
287 break;
288 case SolidUtils::FileType::FileType_Sound:
289 project_assets.audio.push_back(entry.path());
290 break;
291 default:
292 return;
293 }
294 }
295 }

297 inline static void ParseEditorThemes() {
298 themes.clear();
299 for (const auto &entry : std::filesystem::directory_iterator(SolidUtils::GetEditorThemesPath())) {
300 std::string themeName = entry.path().filename().string();
301 std::string themeFile = SolidUtils::GetEditorThemesPath() + "\\" + themeName;
302 pugi::xml_document themeDoc;
303 pugi::xml_parse_result result = themeDoc.load_file(themeFile.c_str());
304
305 if (!result)
306 SolidLogger::GetInstance()->Log(("Error parsing theme: " + themeName).c_str(), themeFile.c_str(), __FILE__, (int*)__LINE__, LogLevel::LogLevel_ERROR);
307
308 pugi::xml_node themeNode = themeDoc.child("theme");
309 Theme theme;
310 theme.name = themeNode.attribute("name").as_string();
311 theme.author = themeNode.attribute("author").as_string();
312 theme.description = themeNode.attribute("description").as_string();
313
314 themes.push_back(theme);
315 }
316 }

1067
1068 char fps[10];

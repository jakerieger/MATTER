68 for (pugi::xml_node color: themeDoc.child("theme").child("colors").children("color")) {

397 float frametime = ImGui::GetIO().Framerate > 0 ? 1000.f / framerate : 0.f;

400 float total_memory = std::get<0>(SolidProfiler::GPU::GetMemoryUsage());
401
402 ImGui::Text("Framerate:");
403 ImGui::SameLine();
404 if (framerate >= 59.f) {
405 ImGui::TextColored(colors["success"], "%.2f", framerate);
406 } else if (framerate >= 30.f) {
407 ImGui::TextColored(colors["warning"], "%.2f", framerate);
408 } else {
409 ImGui::TextColored(colors["error"], "%.2f", framerate);
410 }
411
412 ImGui::Text("Frame Time:");
413 ImGui::SameLine();
414 if (frametime <= 16.94f) {
415 ImGui::TextColored(colors["success"], "%.2f", frametime);
416 } else if (frametime <= 33.33f) {
417 ImGui::TextColored(colors["warning"], "%.2f", frametime);
418 } else {
419 ImGui::TextColored(colors["error"], "%.2f", frametime);
420 }
421
422 float gpu_memory_usage = used_memory / total_memory;
423
424 ImGui::Text("GPU:");
425 ImGui::SameLine();
426 ImGui::Text("%s", gpu_renderer.c_str());
427
428 ImGui::Text("Memory (used):");
429 ImGui::SameLine();
430 if (gpu_memory_usage <= 0.75f) {
431 ImGui::TextColored(colors["success"], "%.2f MB", used_memory / 1000.f);
432 } else if (gpu_memory_usage <= 0.9f) {
433 ImGui::TextColored(colors["warning"], "%.2f MB", used_memory / 1000.f);
434 } else {
435 ImGui::TextColored(colors["error"], "%.2f MB", used_memory / 1000.f);
436 }
437
438 ImGui::Text("Memory (total):");
439 ImGui::SameLine();
440 ImGui::Text("%.2f MB", total_memory / 1000.f);

442 ImGui::End();

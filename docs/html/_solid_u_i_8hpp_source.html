109 inline static void LoadColors(std::string themeName) {
110 std::string defaultTheme = SolidUtils::GetEditorThemesPath() + "\\" + themeName + ".xml";
111 pugi::xml_document themeDoc;
112 pugi::xml_parse_result result = themeDoc.load_file(defaultTheme.c_str());
113
114 if (!result)
115 std::cout << "Failed to load theme file: " << defaultTheme << std::endl;
116
117 for (pugi::xml_node color: themeDoc.child("theme").child("colors").children("color")) {
118 std::string name = color.attribute("name").value();
119 std::string hex = color.attribute("value").value();
120 float alpha = color.attribute("opacity").as_float();
121
122 ImVec4 colorVec = SolidUtils::ColorVecFromHex(hex.c_str(), alpha);
123 colors[name] = colorVec;
124 }
125
126 pugi::xml_node window = themeDoc.child("theme").child("window");
127 window_rounding = window.attribute("window_rounding").as_float();
128 frame_rounding = window.attribute("frame_rounding").as_float();
129 border_size = window.attribute("border_width").as_float();
130 }

222 inline static void ParseProjectAssets(SolidProject &project) {
223 project_assets.textures.clear();
224 project_assets.meshes.clear();
225 project_assets.materials.clear();
226 project_assets.fonts.clear();
227 project_assets.scripts.clear();
228 project_assets.scenes.clear();
229 project_assets.game_objects.clear();
230 project_assets.audio.clear();
231 project_assets.fonts.clear();
232
233 for (const auto &entry : std::filesystem::directory_iterator(project.GetProjectPath())) {
234 SolidUtils::FileType type = SolidUtils::GetFileType(entry.path().extension().string());
235
236 switch (type) {
237 case SolidUtils::FileType::FileType_Font:
238 project_assets.fonts.push_back(entry.path());
239 break;
240 case SolidUtils::FileType::FileType_Script:
241 project_assets.scripts.push_back(entry.path());
242 break;
243 case SolidUtils::FileType::FileType_Scene:
244 project_assets.scenes.push_back(entry.path());
245 break;
246 case SolidUtils::FileType::FileType_Unknown:
247 project_assets.game_objects.push_back(entry.path());
248 break;
249 case SolidUtils::FileType::FileType_Texture:
250 project_assets.textures.push_back(entry.path());
251 break;
252 case SolidUtils::FileType::FileType_Model:
253 project_assets.meshes.push_back(entry.path());
254 break;
255 case SolidUtils::FileType::FileType_Material:
256 project_assets.materials.push_back(entry.path());
257 break;
258 case SolidUtils::FileType::FileType_Sound:
259 project_assets.audio.push_back(entry.path());
260 break;
261 default:
262 return;
263 }
264 }
265 }

862 ImGui::Begin(ICON_FA_STOPWATCH " Analytics");
863 float framerate = ImGui::GetIO().Framerate;
864 float frametime = ImGui::GetIO().Framerate > 0 ? 1000.f / framerate : 0.f;

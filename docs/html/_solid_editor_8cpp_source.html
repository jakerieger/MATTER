109void SolidEditor::ProcessInput() {
110 if (glfwGetMouseButton(mWindow, GLFW_MOUSE_BUTTON_MIDDLE) == GLFW_PRESS) {
111 double dXpos, dYpos;
112 glfwGetCursorPos(mWindow, &dXpos, &dYpos);
113
114 float xpos = static_cast<float>(dXpos);
115 float ypos = static_cast<float>(dYpos);
116
117 if (mFirstMouse) {
118 mLastX = xpos;
119 mLastY = ypos;
120 mFirstMouse = false;
121 }
122
123 float xoffset = xpos - mLastX;
124 float yoffset = mLastY - ypos;
125
126 mLastX = xpos;
127 mLastY = ypos;
128
129 int viewportX, viewportY;
130 glfwGetWindowSize(mWindow, &viewportX, &viewportY);
131 glm::vec2 viewport = glm::vec2(viewportX, viewportY);
132
133 sceneCamera.ProcessMouseOrbit(-xoffset, yoffset, viewport);
134 }

154std::tuple<unsigned int, unsigned int> SolidEditor::CreateRenderBuffers() {
155 unsigned int framebuffer;
156 glGenFramebuffers(1, &framebuffer);
157 glBindFramebuffer(GL_FRAMEBUFFER, framebuffer);
158
159 unsigned int sceneTextureBuffer;
160 glGenTextures(1, &sceneTextureBuffer);
161 glBindTexture(GL_TEXTURE_2D, sceneTextureBuffer);
162 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, WINDOW_WIDTH, WINDOW_HEIGHT, 0, GL_RGB, GL_UNSIGNED_BYTE, NULL);
163 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
164 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
165 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, sceneTextureBuffer, 0);
166
167 unsigned int renderbuffer;
168 glGenRenderbuffers(1, &renderbuffer);
169 glBindRenderbuffer(GL_RENDERBUFFER, renderbuffer);
170 glRenderbufferStorage(GL_RENDERBUFFER, GL_DEPTH24_STENCIL8, WINDOW_WIDTH, WINDOW_HEIGHT);
171 glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_DEPTH_STENCIL_ATTACHMENT, GL_RENDERBUFFER, renderbuffer);
172
173 CheckFramebufferStatus();
174
175 glBindFramebuffer(GL_FRAMEBUFFER, 0);
176
177 mLogger.Log("Created render buffers", "", __FILE__, (int*)__LINE__, LogLevel::LogLevel_INFO);
178
179 return std::make_tuple(framebuffer, sceneTextureBuffer);
180}

182void SolidEditor::CheckFramebufferStatus() {
183 auto fboStatus = glCheckFramebufferStatus(GL_FRAMEBUFFER);
184 if (fboStatus != GL_FRAMEBUFFER_COMPLETE)
185 mLogger.Log("Framebuffer is not complete", "", __FILE__, (int*)__LINE__, LogLevel::LogLevel_ERROR);
186
187 if (fboStatus == GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT)
188 mLogger.Log("Framebuffer is incomplete attachment", "", __FILE__, (int*)__LINE__, LogLevel::LogLevel_ERROR);
189
190 if (fboStatus == GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT)
191 mLogger.Log("Framebuffer is incomplete missing attachment", "", __FILE__, (int*)__LINE__, LogLevel::LogLevel_ERROR);
192
193 if (fboStatus == GL_FRAMEBUFFER_UNSUPPORTED)
194 mLogger.Log("Framebuffer is unsupported", "", __FILE__, (int*)__LINE__, LogLevel::LogLevel_ERROR);
195}

197void SolidEditor::Render() {
198 std::tuple<unsigned int, unsigned int> renderBuffers = CreateRenderBuffers();
199 unsigned int framebuffer = std::get<0>(renderBuffers);
200 unsigned int sceneTexture = std::get<1>(renderBuffers);

203

206 glfwPollEvents();
